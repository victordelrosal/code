Example: Refactoring with AST-grep

# Transform all console.log statements to a proper logger:

ast_grep_replace(
    pattern="console.log($MSG)",
    rewrite="logger.info($MSG)",
    lang="typescript"
)

# Find all async functions that don't handle errors:

ast_grep_search(
    pattern="async function $NAME($$$) { $$$ }",
    lang="typescript"
)

# OpenCode applies changes safely across your entire codebase

Example: Multi-Agent Research & Implementation

# Launch parallel exploration agents in background:

delegate_task(
    subagent_type="explore",
    run_in_background=true,
    prompt="Find authentication patterns in our codebase..."
)

delegate_task(
    subagent_type="librarian",
    run_in_background=true,
    prompt="Find OAuth2 best practices in official docs..."
)

# Consult Oracle for architecture decision:

delegate_task(
    subagent_type="oracle",
    prompt="Analyze trade-offs: JWT vs session-based auth for our scale..."
)

# Delegate implementation with skills:

delegate_task(
    category="visual-engineering",
    load_skills=["frontend-ui-ux", "playwright"],
    prompt="Implement the login flow with OAuth integration..."
)

# Continue session for fixes:

delegate_task(
    session_id="ses_abc123",
    prompt="Fix: Type error on callback handler"
)
